# Years

Years calendar based on an arbitrary input date (to get current year).

## Minimal example

<SourceExample
    example={<DefaultYearsExample/>}
>
```tsx
import {
    Years,
    YearsProvider
}                from "@leight/calendar-client";
import {type FC} from "react";

export interface IDefaultYearsProps {
}

export const DefaultYears: FC<IDefaultYearsProps> = () => {
    return <YearsProvider>
        <Years/>
    </YearsProvider>;
};
```
</SourceExample>
